# 숫자 문자열과 영단어

function solution(s) {
  const numbers = {
    0: "zero",
    1: "one",
    2: "two",
    3: "three",
    4: "four",
    5: "five",
    6: "six",
    7: "seven",
    8: "eight",
    9: "nine",
  };
  let answer = s;

  for (let i = 0; i < 10; i++) {
    let Reg = new RegExp(numbers[i], "g");
    answer = answer.replace(Reg, `${i}`);
  }
  return Number(answer);
}
